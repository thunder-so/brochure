Deploy your [TanStack Start](https://tanstack.com/start) applications to AWS using Thunder patterns. This guide covers static site generation, serverless SSR, and containerized deployments.

There are three deployment patterns available for TanStack Start on AWS:

1. **Static Site Generation (SSG)** — Deploy static TanStack Start sites using S3 and CloudFront with the `Static` construct
2. **Serverless SSR** — Deploy SSR TanStack Start applications using Lambda, S3, and CloudFront with the `TanStackStart` construct
3. **Containerized SSR** — Deploy SSR TanStack Start applications using ECS Fargate with the `Fargate` construct

<Aside type="note" title="Requirements">
- [AWS CDK](https://aws.amazon.com/cdk/) installed globally: `npm install -g aws-cdk`
- Node.js version 20 or higher
- [AWS credentials](https://docs.aws.amazon.com/cli/latest/userguide/cli-configure-files.html) configured
- AWS environment bootstrapped: `cdk bootstrap aws://your-account-id/us-east-1`
</Aside>

## Static Site Generation (SSG) Deployment
---

Deploy static TanStack Start sites to S3 and CloudFront using the `Static` construct. This pattern is ideal for static sites and client-side applications.

### Create Project

<Tabs syncKey="package-manager">
  <TabItem label="npm" icon="seti:npm">
```sh
npx create-tanstack-start my-tanstack-app
cd my-tanstack-app
```
  </TabItem>
  <TabItem label="pnpm" icon="pnpm">
```sh
pnpm create @tanstack/start@latest
cd my-tanstack-app
```
  </TabItem>
  <TabItem label="bun" icon="bun">
```sh
bun create @tanstack/start my-tanstack-app
cd my-tanstack-app
```
  </TabItem>
</Tabs>

### Configure TanStack Start for Static Output

```ts title="app.config.ts"
import { defineConfig } from '@tanstack/start/config';

export default defineConfig({
  server: {
    preset: 'static'
  }
});
```

### Install Dependencies and Setup Stack

<Tabs syncKey="package-manager">
  <TabItem label="npm" icon="seti:npm">
```sh
npm i tsx @thunder-so/thunder --save-dev
```
  </TabItem>
  <TabItem label="pnpm" icon="pnpm">
```sh
pnpm add -D tsx @thunder-so/thunder
```
  </TabItem>
  <TabItem label="bun" icon="bun">
```sh
bun add -d tsx @thunder-so/thunder
```
  </TabItem>
</Tabs>

```ts title="stack/index.ts"
import { Cdk, Static, type StaticProps } from "@thunder-so/thunder";

const myApp: StaticProps = {
  env: {
    account: 'your-account-id',
    region: 'us-east-1'
  },
  application: 'your-application-id',
  service: 'your-service-id',
  environment: 'production',

  rootDir: '',
  outputDir: '.output/public',
};

new Static(
  new Cdk.App(), 
  `${myApp.application}-${myApp.service}-${myApp.environment}-stack`, 
  myApp
);
```

### Deploy

Build and deploy your static site:

<Tabs syncKey="package-manager">
  <TabItem label="npm" icon="seti:npm">
```sh
npm run build
npx cdk deploy --all --app="npx tsx stack/index.ts"
```
  </TabItem>
  <TabItem label="pnpm" icon="pnpm">
```sh
pnpm run build
pnpm exec cdk deploy --all --app="pnpm exec tsx stack/index.ts"
```
  </TabItem>
  <TabItem label="bun" icon="bun">
```sh
bun run build
npx cdk deploy --all --app="bunx tsx stack/index.ts"
```
  </TabItem>
</Tabs>

After deployment, you'll receive a **CloudFront URL** to access your static site.

## Serverless SSR Deployment
---

Deploy server-side rendered TanStack Start applications using Lambda, S3, and CloudFront with the `TanStackStart` construct. This pattern provides auto-scaling SSR with pay-per-request pricing.

### Create Project

<Tabs syncKey="package-manager">
  <TabItem label="npm" icon="seti:npm">
```sh
npx create-tanstack-start my-tanstack-app
cd my-tanstack-app
```
  </TabItem>
  <TabItem label="pnpm" icon="pnpm">
```sh

```
  </TabItem>
  <TabItem label="bun" icon="bun">
```sh
bun create @tanstack/start my-tanstack-app
cd my-tanstack-app
```
  </TabItem>
</Tabs>

### Configure TanStack Start for AWS Lambda

```ts title="app.config.ts"
import { defineConfig } from '@tanstack/start/config';

export default defineConfig({
  server: {
    preset: 'aws-lambda'
  }
});
```

### Install Dependencies and Setup Stack

<Tabs syncKey="package-manager">
  <TabItem label="npm" icon="seti:npm">
```sh
npm i tsx @thunder-so/thunder --save-dev
```
  </TabItem>
  <TabItem label="pnpm" icon="pnpm">
```sh
pnpm add -D tsx @thunder-so/thunder
```
  </TabItem>
  <TabItem label="bun" icon="bun">
```sh
bun add -d tsx @thunder-so/thunder
```
  </TabItem>

```ts title="stack/index.ts"
import { Cdk, TanStackStart, type TanStackStartProps } from "@thunder-so/thunder";

const config: TanStackStartProps = {
  env: {
    account: 'your-account-id',
    region: 'us-east-1'
  },
  application: 'your-application-id',
  service: 'your-service-id',
  environment: 'production',

  rootDir: '',

  serverProps: {
    runtime: Cdk.aws_lambda.Runtime.NODEJS_22_X,
    architecture: Cdk.aws_lambda.Architecture.ARM_64,
    memorySize: 1792,
    timeout: 10,
    keepWarm: true,
  },
};

new TanStackStart(
  new Cdk.App(),
  `${config.application}-${config.service}-${config.environment}-stack`,
  config
);
```

### Environment Variables and Secrets

Configure runtime environment variables and secrets:

```ts title="stack/index.ts"
const config: TanStackStartProps = {
  // ... other props
  
  serverProps: {
    // ... other serverProps
    variables: [
      { NODE_ENV: 'production' },
      { PUBLIC_API_URL: 'https://api.example.com' }
    ],
    secrets: [
      { 
        key: 'DATABASE_URL', 
        resource: 'arn:aws:secretsmanager:us-east-1:123456789012:secret:/my-app/DATABASE_URL-abc123' 
      },
    ],
  },
};
```

```ts title="stack/index.ts"
import { Cdk, Fargate, type FargateProps } from "@thunder-so/thunder";

const svcProps: FargateProps = {
  env: {
    account: 'your-account-id',
    region: 'us-west-2'
  },
  application: 'your-application-id',
  service: 'your-service-id',
  environment: 'production',

  rootDir: '', // e.g. 'app' for monorepos

  new Cdk.App(), 
  `${svcProps.application}-${svcProps.service}-${svcProps.environment}-stack`, 
  svcProps
);
```

### Build Settings Using Nixpacks

Configure automatic containerization with Nixpacks:

```ts title="stack/index.ts"
const svcProps: FargateProps = {
  // ... other props
  
  buildProps: {
    buildSystem: 'Nixpacks',
    installcmd: 'bun install',
    buildcmd: 'bun run build',
    startcmd: 'bun run start',
  },
};
```

### Build Settings Using Docker Container

Alternatively, use a custom Dockerfile:

```dockerfile title="Dockerfile"
FROM public.ecr.aws/docker/library/node:20-alpine AS base

FROM base AS builder
WORKDIR /app

COPY package*.json ./
RUN npm ci

COPY . .
RUN npm run build

FROM base AS runner
WORKDIR /app

ENV NODE_ENV=production

COPY --from=builder /app/dist ./dist

EXPOSE 3000

CMD ["node", "./dist/server/index.js"]
```

```ts title="stack/index.ts"
const svcProps: FargateProps = {
  // ... other props
  
  serviceProps: {
    dockerFile: 'Dockerfile',
    port: 3000,
  },
};
```
